pipeline {
    agent any

    environment {
        CATALINA_HOME = "C:\\apache-tomcat-9.0.109"
    }

    stages {

        stage('Clone Repo') {
            steps {
                git 'https://github.com/archanareddyse/mavenweb.git'
            }
        }

        stage('Build') {
            steps {
                bat "mvn clean package"
            }
        }

        stage('Detect WAR') {
            steps {
                script {
                    def output = bat(
                        script: 'dir /b target\\*.war',
                        returnStdout: true
                    ).trim()

                    def warFile = output.tokenize('\n')[-1].trim()

                    if (!warFile.endsWith(".war")) {
                        error "WAR not found! Output: ${output}"
                    }

                    env.WAR_NAME = warFile
                    echo "WAR detected: ${warFile}"
                }
            }
        }

        stage('Stop Tomcat') {
            steps {
                bat "\"${CATALINA_HOME}\\bin\\shutdown.bat\" || exit 0"
            }
        }

        stage('Deploy') {
            steps {
                bat """
                    copy /Y "target\\${WAR_NAME}" "${CATALINA_HOME}\\webapps\\${WAR_NAME}"
                """
            }
        }

        stage('Start Tomcat') {
            steps {
                bat "\"${CATALINA_HOME}\\bin\\startup.bat\""
            }
        }
    }

    post {
        success {
            echo "üéâ Deployment successful! Visit http://localhost:8080/${WAR_NAME.replace('.war','')}"
        }
        failure {
            echo "‚ùå Deployment failed. Check logs."
        }
    }
}

======================
BASIC GIT SETUP
======================
git --version
git config --global user.name "Your Name"
git config --global user.email "your@email.com"
git config --list

======================
STARTING A REPOSITORY
======================
git init
git clone <repository-url>

======================
CHECK STATUS / HISTORY
======================
git status
git log
git log --oneline
git log --oneline --graph --all
git show <commit-id>

======================
STAGING & COMMITTING
======================
git add .
git add <filename>
git reset <filename>            # unstage file
git commit -m "commit message"
git commit -am "add + commit tracked files"

======================
BRANCHING
======================
git branch
git branch <branch-name>
git checkout <branch-name>
git checkout -b <new-branch>    # create + switch
git branch -d <branch-name>     # delete branch
git branch -D <branch-name>     # force delete

======================
MERGING & REBASING
======================
git merge <branch-name>
git rebase <branch-name>
git rebase --continue
git rebase --abort

======================
PUSHING & PULLING
======================
git remote -v
git remote add origin <repo-url>
git push -u origin <branch-name>
git push
git pull
git fetch

======================
UNDO / FIX MISTAKES
======================
git reset --soft HEAD~1         # undo commit, keep changes staged
git reset --mixed HEAD~1        # undo commit, keep changes unstaged
git reset --hard HEAD~1         # undo commit completely (danger!)
git restore <filename>          # restore file
git restore --staged <file>     # unstage
git checkout -- <filename>      # old restore (legacy)

======================
DELETE FILES / RENAME
======================
git rm <filename>
git rm -r <folder>
git mv <old> <new>

======================
STASH COMMANDS
======================
git stash
git stash save "message"
git stash list
git stash apply
git stash drop
git stash pop

======================
TAGS (RELEASES)
======================
git tag
git tag <tag-name>
git tag -a <tag-name> -m "message"
git push --tags

======================
GIT CLEAN (REMOVE UNTRACKED FILES)
======================
git clean -n                   # preview
git clean -f                   # delete untracked files
git clean -fd                  # delete untracked files + folders

======================
GIT DIFF COMMANDS
======================
git diff
git diff --staged
git diff <branch1> <branch2>

======================
CHECKOUT OLD COMMITS (DETACHED)
======================
git checkout <commit-id>
git switch -                   # go back to previous branch

======================
GITHUB SPECIFIC
======================
git clone <repo-url>
git commit -m "message"
git push
git pull
git pull origin <branch>
git push origin <branch>

======================
BEST PRACTICE SHORTCUTS
======================
git add -A
git commit --amend
git commit --amend --no-edit
git pull --rebase
git push -f                    # force push (danger!)
